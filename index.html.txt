        // ---------- 加载背景大平面 (远处自适应 cover 效果) ----------
        function createBackgroundPlane() {
            const textureLoader = new THREE.TextureLoader();
            const bgTexture = textureLoader.load('background.png', undefined, undefined, (err) => console.warn('background.png加载失败，使用纯色替代', err));
            bgTexture.wrapS = THREE.RepeatWrapping;
            bgTexture.wrapT = THREE.RepeatWrapping;
            
            const geometry = new THREE.PlaneGeometry(1, 1);
            const material = new THREE.MeshStandardMaterial({ map: bgTexture, side: THREE.DoubleSide, emissive: 0x111111, emissiveIntensity: 0.3 });
            bgPlane = new THREE.Mesh(geometry, material);
            scene.add(bgPlane);
            updateBackgroundScale(); // 根据相机视野调整平面大小和位置
        }
        
        function updateBackgroundScale() {
            if (!bgPlane) return;
            // 相机位置 (0,2,12), 背景放在 Z = -25 处
            const distance = 28;
            const vFov = (camera.fov * Math.PI) / 180;
            const planeHeight = 2 * Math.tan(vFov / 2) * distance;
            const planeWidth = planeHeight * camera.aspect;
            bgPlane.scale.set(planeWidth, planeHeight, 1);
            bgPlane.position.set(0, 1, -28);
            // 稍微旋转使其正对相机 (已经是面向相机方向，平面默认面向Z正轴，相机在Z正方向，直接放置即可)
        }
        
        // ---------- 加载 GLTF 模型 ----------
        const loader = new GLTFLoader();
        loader.load('model.glb', 
            (gltf) => {
                mainModel = gltf.scene;
                mainModel.position.set(0, -7, -25);
                mainModel.traverse((child) => {
                    if (child.isMesh) {
                        child.castShadow = true;
                        child.receiveShadow = false;
                    }
                });
                scene.add(mainModel);
                modelLoaded = true;
                // 提取模型顶点用于粒子效果 (遍历所有Mesh几何体，合并顶点)
                const tempPositions = [];
                mainModel.traverse((child) => {
                    if (child.isMesh && child.geometry) {
                        const geom = child.geometry;
                        if (geom.attributes.position) {
                            const posAttr = geom.attributes.position.array;
                            for (let i = 0; i < posAttr.length; i += 3) {
                                // 将局部坐标转换为世界坐标近似 (因为模型还未变换, 临时提取局部坐标)
                                // 实际消散时使用模型当前世界矩阵变换，这里储存相对模型的局部坐标，粒子生成时转换
                                tempPositions.push(new THREE.Vector3(posAttr[i], posAttr[i+1], posAttr[i+2]));
                            }
                        }
                    }
                });
                modelGeometryVertices = tempPositions;
                console.log(`模型加载成功，顶点数: ${modelGeometryVertices.length}`);
                // 初始状态 IDLE
                resetToIdle();
            },
            undefined,
            (error) => {
                console.error('model.glb 加载失败，请确保文件存在同级目录', error);
                // 创建一个替代占位模型以便演示交互 (保证功能可见)

                const material = new THREE.MeshStandardMaterial({ color: 0xaa88ff, roughness: 0.3, metalness: 0.7 });
                mainModel = new THREE.Mesh(geometry, material);

                scene.add(mainModel);
                modelLoaded = true;
                // 生成简单顶点数据
                const posAttr = geometry.attributes.position.array;
                const tempVerts = [];
                for (let i = 0; i < posAttr.length; i += 3) {
                    tempVerts.push(new THREE.Vector3(posAttr[i], posAttr[i+1], posAttr[i+2]));
                }
                modelGeometryVertices = tempVerts;
                resetToIdle();
            }

        // ---------- 辅助函数: 更新模型Y轴 (仅ASCENDING状态下手势控制) ----------
        function updateModelYFromHand() {
            if (!mainModel) return;
            if (currentState === 'ASCENDING' && handLandmarks && handLandmarks.length > 0) {
                // 使用手腕或者掌心点(0)的y坐标，归一化范围(0~1) -> 控制模型Y
                let palmY = handLandmarks[0].y; // 0~1 顶部为0底部为1
                // 手越高(y越小) 模型越高 映射: y=0 -> MAX_MODEL_Y; y=1 -> MIN_MODEL_Y
                let mappedY = MIN_MODEL_Y + (1 - palmY) * (MAX_MODEL_Y - MIN_MODEL_Y);
                mappedY = Math.min(MAX_MODEL_Y, Math.max(MIN_MODEL_Y, mappedY));
                targetModelY = mappedY;
                mainModel.position.y = targetModelY;
            } else if (currentState === 'ASCENDING' && !handLandmarks) {
                // 没有手部数据时保持不动
            }
        }
        
        // 重置整个游戏状态 (IDLE, 模型归位, 清除粒子, 停止浮动)
        function resetToIdle() {
            if (hoverTween) { hoverTween.kill(); hoverTween = null; }
            if (particleSystem) {
                scene.remove(particleSystem);
                if (particleSystem.geometry) particleSystem.geometry.dispose();
                if (particleSystem.material) particleSystem.material.dispose();
                particleSystem = null;
            }
            if (particleAnimationId) {
                cancelAnimationFrame(particleAnimationId);
                particleAnimationId = null;
            }
            if (mainModel) {
                mainModel.visible = true;
                // 重置位置和旋转
                mainModel.position.set(0, -7, -25);
                mainModel.rotation.set(0, 0, 0);
                // 如果有额外动画停止
                gsap.killTweensOf(mainModel.position);
            }
            currentState = 'IDLE';
            updateStatusText('🟢 IDLE | 张开手掌召唤');
            // 清除浮动标志
        }
        
        // 阶段2: 悬停动画 (快速移动到前方并上下浮动)
        function startHovering() {
            if (!mainModel) return;
            if (hoverTween) hoverTween.kill();
            // 使用 GSAP 移动模型位置到 (0, 当前Y, -8) 保持当前Y，但要求移动到(0,0,-8)，但为了悬停，我们使用当前Y可能不是0，但是为了体验，平滑到(0, mainModel.position.y, -8)
            const targetPos = { x: 0, y: mainModel.position.y, z: -8 };
            gsap.to(mainModel.position, {
                duration: 0.8,
                x: targetPos.x,
                z: targetPos.z,
                ease: "power2.out",
                onComplete: () => {
                    // 完成位移后，开始上下浮动 (轻微)
                    if (currentState === 'HOVERING_NEAR') {
                        hoverTween = gsap.to(mainModel.position, {
                            duration: 1.2,
                            y: mainModel.position.y + 0.25,
                            repeat: -1,
                            yoyo: true,
                            ease: "sine.inOut",
                        });
                    }
                }
            });
        }
        
        // 阶段3: 灰烬消散效果 (基于顶点生成粒子，向外向上发散，逐渐透明)
        function startDissolving() {
            if (!mainModel || !modelGeometryVertices || modelGeometryVertices.length === 0) {
                console.warn("无法生成粒子：无顶点数据");
                resetToIdle();
                return;
            }
            // 隐藏原模型
            mainModel.visible = false;
            // 获取模型当前世界矩阵，用于转换粒子初始位置
            const matrixWorld = mainModel.matrixWorld;
            const positions = [];
            const velocities = [];
            const sizes = [];
            // 采样顶点，数量较多 (最多15000个粒子)
            let maxParticles = Math.min(modelGeometryVertices.length, 12000);
            const step = Math.max(1, Math.floor(modelGeometryVertices.length / maxParticles));
            for (let i = 0; i < modelGeometryVertices.length; i += step) {
                const localPos = modelGeometryVertices[i];
                const worldPos = localPos.clone().applyMatrix4(matrixWorld);
                positions.push(worldPos.x, worldPos.y, worldPos.z);
                // 随机速度: 向外辐射 + 向上飘散
                const outwardDir = new THREE.Vector3(worldPos.x, worldPos.y * 0.5, worldPos.z + 2).normalize();
                const upDir = new THREE.Vector3(0, 1, 0);
                const vel = outwardDir.multiplyScalar(0.08 + Math.random() * 0.12).add(upDir.multiplyScalar(0.05 + Math.random() * 0.1));
                velocities.push(vel);
                // 随机大小 0.08~0.32
                sizes.push(0.08 + Math.random() * 0.24);
            }
            const particleCount = positions.length / 3;
            console.log(`生成粒子数量: ${particleCount}`);
            
            // 创建BufferGeometry并加入每个粒子的大小属性 (用于着色器)
            const particleGeo = new THREE.BufferGeometry();
            particleGeo.setAttribute('position', new THREE.BufferAttribute(new Float32Array(positions), 3));
            particleGeo.setAttribute('size', new THREE.BufferAttribute(new Float32Array(sizes), 1));
            
            // 自定义着色器材质 (支持每个粒子大小和整体透明度渐隐)
            const vertexShader = `
                attribute float size;
                varying float vAlpha;
                void main() {
                    vAlpha = 1.0;
                    vec4 mvPosition = modelViewMatrix * vec4(position, 1.0);
                    gl_PointSize = size * ( 300.0 / ( - mvPosition.z ) );
                    gl_PointSize = clamp(gl_PointSize, 4.0, 28.0);
                    gl_Position = projectionMatrix * mvPosition;
                }
            `;
            const fragmentShader = `
                uniform sampler2D pointTexture;
                uniform float uAlpha;
                varying float vAlpha;
                void main() {
                    vec4 texColor = texture2D(pointTexture, gl_PointCoord);
                    float finalAlpha = texColor.a * uAlpha;
                    gl_FragColor = vec4(texColor.rgb, finalAlpha * 0.9);
                }
            `;
            const canvas = document.createElement('canvas');
            canvas.width = 32;
            canvas.height = 32;
            const ctx = canvas.getContext('2d');
            ctx.fillStyle = '#ffaa66';
            ctx.beginPath();
            ctx.arc(16, 16, 12, 0, 2*Math.PI);
            ctx.fill();
            ctx.globalCompositeOperation = 'source-over';
            ctx.fillStyle = '#cc8844';
            ctx.beginPath();
            ctx.arc(16, 16, 6, 0, 2*Math.PI);
            ctx.fill();
            const texture = new THREE.CanvasTexture(canvas);
            
            const particleMaterial = new THREE.ShaderMaterial({
                uniforms: {
                    pointTexture: { value: texture },
                    uAlpha: { value: 1.0 }
                },
                vertexShader: vertexShader,
                fragmentShader: fragmentShader,
                transparent: true,
                depthWrite: false,
                blending: THREE.AdditiveBlending
            });
            
            particleSystem = new THREE.Points(particleGeo, particleMaterial);
            scene.add(particleSystem);
            
            // 动画消散: 每帧更新粒子位置和整体透明度
            let startTime = performance.now();
            const duration = 3500; // 3.5秒消散
            
            function updateParticles(now) {
                const elapsed = now - startTime;
                let progress = Math.min(1, elapsed / duration);
                // 更新透明度
                particleMaterial.uniforms.uAlpha.value = Math.max(0, 1 - progress * 1.2);
                // 更新每个粒子的位置: 向外向上飘散
                const positionsAttr = particleSystem.geometry.attributes.position.array;
                for (let i = 0; i < particleCount; i++) {
                    const idx = i * 3;
                    const vel = velocities[i];
                    // 随时间增加速度略微加快
                    const speedFactor = 1 + progress * 0.8;
                    positionsAttr[idx] += vel.x * 0.25 * speedFactor;
                    positionsAttr[idx+1] += vel.y * 0.3 * speedFactor + 0.02;
                    positionsAttr[idx+2] += vel.z * 0.2 * speedFactor;
                }
                particleSystem.geometry.attributes.position.needsUpdate = true;
                
                if (progress < 1) {
                    particleAnimationId = requestAnimationFrame(updateParticles);
                } else {
                    // 动画结束，清除粒子系统，重置状态到IDLE
                    scene.remove(particleSystem);
                    if (particleSystem.geometry) particleSystem.geometry.dispose();
                    if (particleSystem.material) particleSystem.material.dispose();
                    particleSystem = null;
                    resetToIdle();
                    particleAnimationId = null;
                }
            }
            if (particleAnimationId) cancelAnimationFrame(particleAnimationId);
            particleAnimationId = requestAnimationFrame(updateParticles);
        }

        // ---------- 手势识别逻辑 (张开手掌 / 握拳) ----------
        function analyzeGesture(landmarks) {
            if (!landmarks || landmarks.length === 0) return { isFist: false, isOpen: false };
            // 手指关键点索引: 指尖: 4(拇指),8(食指),12(中指),16(无名指),20(小指)
            // 指根 (MCP): 1,5,9,13,17 (近似)
            const tips = [4, 8, 12, 16, 20];
            const bases = [2, 5, 9, 13, 17];
            let extendedCount = 0;
            for (let i = 0; i < tips.length; i++) {
                const tip = landmarks[tips[i]];
                const base = landmarks[bases[i]];
                const dist = Math.hypot(tip.x - base.x, tip.y - base.y, tip.z - base.z);
                // 归一化距离阈值 (因为手部大小大致固定，且坐标范围0-1)
                if (dist > 0.045) extendedCount++;
            }
            // 张开手掌: 至少4个手指伸展
            const isOpen = extendedCount >= 4;
            // 握拳: 伸展手指少于2个且中指弯曲明显
            const isFist = extendedCount <= 1;
            return { isFist, isOpen };
        }
        
        // MediaPipe Hands 初始化
        function initMediaPipe() {
            videoElement = document.getElementById('cameraPreview');
            hands = new Hands({
                locateFile: (file) => {
                    return `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`;
                }
            });
            hands.setOptions({
                maxNumHands: 1,
                modelComplexity: 1,
                minDetectionConfidence: 0.5,
                minTrackingConfidence: 0.5
            });
            hands.onResults((results) => {
                if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) {
                    handLandmarks = results.multiHandLandmarks[0];
                    isHandDetected = true;
                    const gesture = analyzeGesture(handLandmarks);
                    isFist = gesture.isFist;
                    isOpenPalm = gesture.isOpen;
                } else {
                    handLandmarks = null;
                    isHandDetected = false;
                    isFist = false;
                    isOpenPalm = false;
                }
            });
            
            // 启动摄像头
            cameraInstance = new Camera(videoElement, {
                onFrame: async () => {
                    if (hands) await hands.send({ image: videoElement });
                },
                width: 640,
                height: 480
            });
            cameraInstance.start();
        }
        
        // ---------- 游戏状态机 (每帧根据手势更新) ----------
        function updateGameLogic() {
            if (!modelLoaded || !mainModel) return;
            
            // 根据当前状态处理转换
            if (currentState === 'IDLE') {
                if (isOpenPalm && isHandDetected) {
                    currentState = 'ASCENDING';
                    updateStatusText('📈 ASCENDING | 举手控制高度');
                    // 确保模型处于上升起始位置 (如果不在范围)
                    if (mainModel.position.y < -5) mainModel.position.y = -7;
                }
            }
            else if (currentState === 'ASCENDING') {
                // 手势控制模型Y
                updateModelYFromHand();
                // 检查是否接近最高高度并且握拳
                const currentY = mainModel.position.y;
                if (currentY >= ASCEND_TRIGGER_HIGH && isFist && isHandDetected) {
                    // 进入悬停阶段
                    currentState = 'HOVERING_NEAR';
                    updateStatusText('🖐️ HOVERING_NEAR | 握拳触发靠近');
                    startHovering();
                }
                // 如果丢失手势或者张开变弱，但保持在ASCENDING，没有额外动作
            }
            else if (currentState === 'HOVERING_NEAR') {
                // 检测到张开手掌触发消散
                if (isOpenPalm && isHandDetected) {
                    currentState = 'DISSOLVING';
                    updateStatusText('💨 DISSOLVING | 灰烬消散...');
                    if (hoverTween) { hoverTween.kill(); hoverTween = null; }
                    // 停止浮动后开始粒子消散
                    startDissolving();
                }
            }
            else if (currentState === 'DISSOLVING') {
                // 消散中不做手势交互，等待动画重置
                // 为防止重复触发，忽略手势输入
                return;
            }
            
            // 对于ASCENDING状态如果手部丢失，模型暂时保持当前高度，不做额外操作，但是依然可以控制一旦恢复
        }
        
        // 重置按钮逻辑
        document.getElementById('resetBtn').addEventListener('click', () => {
            if (currentState === 'DISSOLVING') {
                // 强制中断消散动画并重置
                if (particleAnimationId) cancelAnimationFrame(particleAnimationId);
                if (particleSystem) scene.remove(particleSystem);
                particleSystem = null;
            }
            if (hoverTween) { hoverTween.kill(); hoverTween = null; }
            resetToIdle();
        });
        
        // 全屏按钮
        document.getElementById('fullscreenBtn').addEventListener('click', () => {
            if (!document.fullscreenElement) {
                document.documentElement.requestFullscreen();
            } else {
                document.exitFullscreen();
            }
        });
        
        // Camera显隐按钮
        const camPreview = document.getElementById('cameraPreview');
        document.getElementById('toggleCamBtn').addEventListener('click', () => {
            cameraVisible = !cameraVisible;
            camPreview.style.opacity = cameraVisible ? '1' : '0';
            camPreview.style.visibility = cameraVisible ? 'visible' : 'hidden';
        });
        
        // 窗口适配
        window.addEventListener('resize', onWindowResize, false);
        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
            if (bgPlane) updateBackgroundScale();
        }
        
        // 动画循环渲染
        function animateRender() {
            requestAnimationFrame(animateRender);
            // 更新游戏状态逻辑
            updateGameLogic();
            // 让背景平面始终保持面向相机(简易版，平面无需旋转因为已经固定在Z轴)
            if (bgPlane) {
                // 可微调面向相机但没必要，因为平面足够大且固定，透视cover效果已经满足
            }
            // 让灯光有微小动态增强立体感
            const time = Date.now() * 0.002;
            rimLight.intensity = 0.55 + Math.sin(time) * 0.1;
            // 相机轻微注视模型中央
            if (mainModel && currentState !== 'DISSOLVING') {
                // 让相机注视模型位置 (平滑)
                const targetPos = mainModel.position.clone();
                if (currentState === 'HOVERING_NEAR') {
                    camera.lookAt(targetPos);
                } else {
                    camera.lookAt(0, Math.max(-2, mainModel.position.y * 0.5), -5);
                }
            } else {
                camera.lookAt(0, 0, -5);
            }
            renderer.render(scene, camera);
        }
        
        // 初始化场景、背景、启动所有系统
        function init() {
            createBackgroundPlane();
            initMediaPipe();
            animateRender();
            // 每帧额外让相机视角平滑
        }
        
        init();
        
        // 控制台友好提示
        console.log('游戏已启动，请确保 model.glb 和 background.png 与 index.html 同级');
    </script>
